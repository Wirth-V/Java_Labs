public class MyList<T> {
public MyList() {
super();
}

T data;

MyList head;
MyList tail;
int n;
void add_first(T data) { //add first
MyList a = new MyList();
a.data = data;
if (head == null)
{
head = a;
tail = a;
} else {
a.next = head;
head = a;
}
n++;
}

}

MyList ML1= new MyList();
T tmp1 = (T)array[1].data;
T tmp2 = (T)array[0].data;
int res = tmp1.compareTo(tmp2);
for (int i =0 ; i < k; i++) {
for(int j=i;j>0; j--) {
tmp1 = (T)array[j].data;
tmp2 = (T)array[j-1].data;
res = tmp1.compareTo(tmp2);
if(res<0) {
MyList tmp=array[j-1];
array[j-1]=array[j];
array[j]=tmp;
}
}
}

for(int o = k-1;o>-1;o--) {
ML1.add_first(array[o].data);
}
return ML1;
}
